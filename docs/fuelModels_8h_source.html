87 struct FuelModelRecord
88 {
89 int fuelModelNumber_; // Standard ID number for fuel model
90 std::string code_; // Fuel model code, usually 2 letters followed by number,(e.g., "GR1")
91 std::string name_; // Fuel model name, (e.g., "Humid Climate Grass")
92 double fuelbedDepth_; // Fuelbed depth in feet
93 double moistureOfExtinctionDead_; // Dead fuel extinction moisture content (fraction)
94 double heatOfCombustionDead_; // Dead fuel heat of combustion (Btu/lb)
95 double heatOfCombustionLive_; // Live fuel heat of combustion (Btu/lb)
96 double fuelLoadOneHour_; // Dead 1 hour fuel loading (lb/ft^2)
97 double fuelLoadTenHour_; // Dead 10 hour fuel loading (lb/ft^2)
98 double fuelLoadHundredHour_; // Dead 100 hour fuel loading (lb/ft^2)
99 double fuelLoadLiveHerbaceous_; // Live herb fuel loading (lb/ft^2)
100 double fuelLoadLiveWoody_; // Live wood fuel loading (lb/ft^2)
101 double savrOneHour_; // Dead 1-h fuel surface area to volume ratio (ft^2/ft^3)
102 double savrLiveHerbaceous_; // Live herb surface area to volume ratio (ft^2/ft^3)
103 double savrLiveWoody_; // Live wood surface area to volume ratio (ft^2/ft^3)
104 bool isDynamic_; // If true, the fuel model is dynamic
105 bool isReserved_; // If true, record cannot be used for custom fuel model
106 bool isDefined_; // If true, record has been populated with values for its fields
107 };
